\newcommand{\ans}[2]{%
	\overset{\textcolor{green!70!black}{#2}}{\rule{#1}{0.4pt}}%
}
\newcommand{\ansqus}[2]{%
    \overset{\textcolor{green!70!black}{#1}}{\rule{1cm}{0.4pt}}
    +\overset{\textcolor{green!70!black}{#2}}{\rule{1cm}{0.4pt}}%
}
\newcommand{\ansdia}[2]{%
	\node[below, text width=#1, align=center] at (current bounding box.south) {%
		$\overset{\textcolor{green!70!black}{#2}}{\rule{#1}{0.4pt}}$%
	};%
}

\NewDocumentCommand{\triangleDiagram}{O{0} O{0} O{4} O{0} O{2} O{3}}{%
  \tkzDefPoint(#1,#2){A}
  \tkzDefPoint(#3,#4){C}
  \tkzDefPoint(#5,#6){B}
  \tkzDrawPolygon(A,B,C)
}

\newcommand{\extendArrow}[4]{%
  \tkzDefPointWith[linear,K=#3](#1,#2)\tkzGetPoint{#4}%
  \tkzDrawSegment[-Triangle](#1,#4)%
}

\newcommand{\markAngle}[4]{% Used Mark Angle with Label
  \tkzMarkAngle[size=0.6](#2,#3,#4)%
  \tkzLabelAngle[pos=1.0](#2,#3,#4){#1}%
}
