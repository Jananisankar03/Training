\newcommand{\splitdigits}[1]{%
	\def\result{}%
	\splitloop#1\relax
	\result
}
\def\splitloop#1{%
	\ifx#1\relax
	\else
	\ifx\result\empty
	\def\result{#1}%
	\else
	\edef\result{\result\kern1cm#1}
	\fi
	\expandafter\splitloop
	\fi
}
